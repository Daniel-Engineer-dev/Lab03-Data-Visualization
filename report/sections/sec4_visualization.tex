\section{Phân tích Dữ liệu và Trực quan hóa}
\label{sec:visualization}

Chương này trình bày sườn phân tích và các biểu đồ dự kiến cho bốn mục tiêu đã xác định ở Chương~\ref{sec:analysis_goals}. Để báo cáo ngắn gọn, mỗi mục tiêu chỉ sử dụng hai biểu đồ chính, được ghép chung trong một hình minh họa và đi kèm các nhận xét quan trọng nhất.

\subsection{Tổng quan và xu hướng thời gian}

\textbf{Thành viên phụ trách}: Lê Lâm Trí Đức.

\textbf{Mục tiêu}: Trình bày bức tranh tổng quan về số vụ va chạm trong giai đoạn 2020--2024; đồng thời xác định các thời điểm có tần suất tai nạn cao.

\textbf{Biểu đồ thực tế}:
\begin{itemize}
    \item Biểu đồ đường (Line chart): Thể hiện xu hướng số vụ va chạm phân theo mức độ nghiêm trọng (Nhẹ, Nghiêm trọng, Tử vong) giai đoạn 2020--2024.
    \item Biểu đồ bản đồ nhiệt (Heatmap Matrix): Thể hiện mật độ số vụ va chạm theo ngày trong tuần (Thứ) và khung giờ trong ngày (0h -- 23h).
\end{itemize}

\textbf{Nhận xét rút ra từ dữ liệu}:
\begin{itemize}
    \item \textbf{Xu hướng biến động theo năm}: Tổng số vụ va chạm có xu hướng tăng mạnh từ năm 2020 (đạt 91,2 nghìn vụ) và đạt đỉnh vào năm 2022 với 106,0 nghìn vụ. Sau năm 2022, số vụ va chạm bắt đầu có xu hướng giảm nhẹ về năm 2023 (104,3 nghìn vụ) và 2024 (100,9 nghìn vụ). Xu hướng này chủ yếu được chi phối bởi các vụ va chạm ở mức độ Nhẹ (Slight) chiếm tỷ trọng lớn nhất. Trong khi đó, các vụ tai nạn ở mức Nghiêm trọng (Serious) và Tử vong (Fatal) có xu hướng đi ngang hoặc tăng nhẹ qua các năm.
    \item \textbf{Mật độ va chạm theo khung giờ}: Tai nạn giao thông chủ yếu tập trung vào ban ngày và đặc biệt tăng mạnh vào khung giờ tan tầm buổi chiều từ 15h00 đến 18h00 (đỉnh điểm lúc 16h00 -- 17h00) do lưu lượng giao thông đông đúc, tần suất đi lại tăng cao khi tan học và tan làm.
    \item \textbf{Mật độ va chạm theo ngày trong tuần}: Tần suất xảy ra tai nạn giao thông vào các ngày làm việc trong tuần (Thứ Hai đến Thứ Sáu) cao hơn rõ rệt so với hai ngày cuối tuần (Thứ Bảy và Chủ Nhật). Sự kết hợp giữa ngày trong tuần và giờ trong ngày trên bản đồ nhiệt chỉ ra rằng, khoảng thời gian từ 16h00 đến 17h00 các ngày trong tuần là thời điểm có mật độ va chạm cao nhất cả tuần (thể hiện qua dải màu đỏ đậm nhất).
\end{itemize}

\subsection{Điều kiện đường sá và môi trường}

\textbf{Thành viên phụ trách}: Hoàng Quốc Việt.

\textbf{Mục tiêu}: So sánh mức độ nghiêm trọng của tai nạn theo các điều kiện xảy ra va chạm, gồm loại đường, giới hạn tốc độ, ánh sáng, thời tiết và tình trạng mặt đường.

\textbf{Biểu đồ đề xuất}:
\begin{itemize}
    \item Stacked bar chart: số vụ va chạm theo \texttt{collision\_severity} và \texttt{road\_type}.
    \item Matrix hoặc clustered bar chart: mức độ nghiêm trọng theo thời tiết và điều kiện ánh sáng.
\end{itemize}

\textbf{Nhận xét cần rút ra}: trình bày từ ba đến bốn nhận xét về loại đường hoặc điều kiện môi trường thường xuất hiện trong các vụ tai nạn nghiêm trọng.

\subsection{Phương tiện và người lái}

\textbf{Thành viên phụ trách}: Nguyễn Lê Thế Vinh.

\textbf{Mục tiêu}: Phân tích đặc điểm phương tiện và người lái trong các vụ va chạm, từ đó nhận diện nhóm phương tiện hoặc nhóm người lái xuất hiện nhiều trong các vụ tai nạn nghiêm trọng.

\textbf{Biểu đồ đề xuất}:
\begin{itemize}
    \item Bar chart: số phương tiện liên quan theo \texttt{vehicle\_type}.
    \item Stacked column chart: mức độ nghiêm trọng theo nhóm tuổi người lái.
\end{itemize}

\textbf{Nhận xét cần rút ra}: trình bày từ ba đến bốn nhận xét về loại phương tiện chiếm tỷ trọng cao và nhóm tuổi người lái cần chú ý.

\subsection{Người thương vong và nhóm rủi ro}

\textbf{Thành viên phụ trách}: Phạm Quang Vinh.

\textbf{Mục tiêu}: Phân tích đặc điểm người thương vong theo độ tuổi, giới tính, vai trò tham gia giao thông và mức độ thương vong để xác định các nhóm rủi ro cao.

\textbf{Biểu đồ đề xuất}:
\begin{itemize}
    \item Stacked bar chart: số người thương vong theo \texttt{casualty\_severity} và nhóm tuổi.
    \item Bar chart: số người thương vong theo vai trò tham gia giao thông.
\end{itemize}

\textbf{Nhận xét cần rút ra}: trình bày từ ba đến bốn nhận xét về nhóm tuổi hoặc vai trò có mức độ thương vong cao và nhóm đối tượng nên được ưu tiên trong khuyến nghị an toàn.
